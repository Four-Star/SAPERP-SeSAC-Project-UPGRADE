MODULE set_alv_0100 OUTPUT.
  IF go_con_0100 IS INITIAL .
    PERFORM create_object USING 'CON1'
                          CHANGING go_con_0100 go_alv_0100 .
    PERFORM set_layo_0100 CHANGING gs_layo_0100 .
    PERFORM set_sort_0100 CHANGING gt_sort_0100 .
    PERFORM set_fcat_0100 CHANGING gt_fcat_0100 .
  ENDIF.

  PERFORM set_handler USING go_alv_0100
                      CHANGING go_handler .
  PERFORM display_alv_0100 USING go_alv_0100 gs_layo_0100 gt_sort_0100 gt_fcat_0100 'ZSFS0402'
                           CHANGING gt_data .

*&---------------------------------------------------------------------*
*& Module MODIFY_SCREEN_0200 OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE modify_screen_0200 OUTPUT.
  LOOP AT SCREEN .
    IF gv_step = 0 .
      " ### ####
      IF screen-group1 = 'HD' .
        screen-input = '1' .
      ELSEIF screen-group1 = 'LN' OR screen-name = 'CON2' .
        screen-active = '0' .
*        screen-invisible = 1.
      ENDIF.
    ELSEIF gv_step = 1 .
      " #### ####
      IF screen-group1 = 'HD' .
        screen-input = '0' .
      ELSEIF screen-group1 = 'LN' OR screen-name = 'CON2' .
        screen-active = '1' .
*        screen-invisible = 0.
      ENDIF.

      IF screen-group3 = 'IT' .
        IF zsfs0403-btype = 'KZ' or zsfs0403-btype = 'DZ' .
          SCREEN-ACTIVE = 0 .
        ENDIF.
      ENDIF.
    ENDIF.

    MODIFY SCREEN .
  ENDLOOP.
ENDMODULE.

*& Module MODIFY_SCREEN_0201 OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE modify_screen_0201 OUTPUT.
  LOOP AT SCREEN .
    IF gv_step_sub = 0 .
      " ###
      IF screen-group1 = 'ST1' .
        " ### ## # ##
        screen-active = 1 .
      ELSEIF screen-group1 = 'ST2' .
        " ## ## ###
        screen-active = 0 .
      ENDIF.
    ELSEIF gv_step_sub = 1 .
      " ###
      IF screen-group1 = 'ST1' .
        screen-active = 1.
        IF screen-group2 = 'BSC' .
          screen-input = 0 .
        ENDIF.
      ELSEIF screen-group1 = 'ST2' .
        " ## #### ### PARTNER ### ## ## ### ## ##
        " ### ##
        screen-active = 1.
        IF gv_tmp <> 'X' AND screen-group2 = 'BP' .  " BP# #### ##
          screen-active = 0 .
        ELSEIF gv_tmp = 'X' AND screen-group2 = 'BP' . " BP# ###
          screen-input = 0 .
        ENDIF.

      ENDIF.

    ENDIF.

    MODIFY SCREEN .
  ENDLOOP.
ENDMODULE.

*&---------------------------------------------------------------------*
*& Module MODIFY_SCREEN_0202 OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE modify_screen_0202 OUTPUT.
  LOOP AT SCREEN .
    CASE zsfs0403-btype .
      WHEN 'KZ' . " ##
        IF SCREEN-group1 = 'VD' .
          SCREEN-ACTIVE = 1 .
        ELSEIF SCREEN-group1 = 'CT' .
          SCREEN-ACTIVE = 0 .
        ENDIF.
      WHEN 'DZ' . " ##
        IF SCREEN-group1 = 'VD' .
          SCREEN-ACTIVE = 0 .
        ELSEIF SCREEN-group1 = 'CT' .
          SCREEN-ACTIVE = 1 .
        ENDIF.
      WHEN OTHERS.
    ENDCASE.

    MODIFY SCREEN .
  ENDLOOP.

ENDMODULE.
